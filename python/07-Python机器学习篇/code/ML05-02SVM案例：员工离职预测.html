In [28]:
best_score=0 #将最优精度的初始值设置为0

for gamma in [0.001,0.01,0.1,1,10]:
    for C in [0.01,0.1,1,10,100]:
        #对每种参数组合都训练一个SVC
        svm=SVC(gamma=gamma,C=C)
         #训练集用于训练模型
        svm.fit(X_train,y_train)
        #在测试集上评估SVC
        score=svm.score(X_test,y_test) #验证集用于评估所选参数的性能
        #如果得到了更高的分数，则保存该分数和对应的参数
        if score>best_score:
            best_score=score
            best_parameters={'C':C,'gamma':gamma}

#为SVC这个函数设置字典格式的参数
svm=SVC(**best_parameters) 
svm.fit(X_train,y_train) #利用训练集训练模型
test_score=svm.score(X_test,y_test) #利用测试集计算精度

print("Best score on validation set:{:.3f}".format(best_score))
print("Best parameters:{}".format(best_parameters))
print("Test set score with best parameters:{:.3f}".format(test_score))

from sklearn.model_selection import KFold
from sklearn.model_selection import cross_val_score

best_score=0

for gamma in [0.001,0.01,0.1,1,10]:
    for C in [0.01,0.1,1,10,100]:
        #对于每种参数组合都训练一个SVC
        svm=SVC(gamma=gamma,C=C)
        svm.fit(X_trainval,y_trainval)
        #执行交叉验证
        kfolder=KFold(n_splits=5,shuffle=True,random_state=0)
        scores=cross_val_score(svm,X_trainval,y_trainval,cv=kfolder)
        #计算交叉验证平均精度
        score=scores.mean()
        #如果我们得到了更高的分数，则保存分数和对应的参数
        if score>best_score:
            best_score=score
            best_parameters={'C':C,'gamma':gamma}

svm=SVC(**best_parameters)
svm.fit(X_trainval,y_trainval)
test_score=svm.score(X_test,y_test)

print("Best score on validation set:{:.3f}".format(best_score))
print("Best parameters:{}".format(best_parameters))
print("Test set score with best parameters:{:.3f}".format(test_score))
