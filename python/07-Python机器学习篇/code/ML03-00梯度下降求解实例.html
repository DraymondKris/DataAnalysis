def gradient_descent_f1(X,learning_rate,stop_error):
    losses=[lost_function(X)]
    count=0

    while True:
        X=X-learning_rate*gradient_f(X)
        losses.append(lost_function(X))
        count+=1
        
        if abs(losses[-1]-losses[-2])<stop_error:
            break
        
    return X,count,losses

def gradient_descent_f2(X,learning_rate,max_loop):
    losses=[]
    
    for i in range(max_loop):
        X = X - learning_rate * gradient_f(X)
        loss=lost_function(X)
        losses.append(loss)
    
    return X,losses
